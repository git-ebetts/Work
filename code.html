var capture;
var imgA;
var imgB;
var imgC;
var imgD;

var img1;
var img2;
var img3;
var img4;

var imgMaskA;
var imgMaskC;
var imgMask2;
var imgMask4;
var loc;

function preload() {

imgMaskA = loadImage("assets/mask_a.png");
imgMaskC = loadImage("assets/mask_c.png");
imgMask2 = loadImage("assets/mask_2.png");
imgMask4 = loadImage("assets/mask_4.png");
}

function setup() {

createCanvas(640, 480);
capture = createCapture(VIDEO);
capture.size(320, 240);
capture.hide();
imgA = createImage(320, 240);
imgB = createImage(320, 240);
imgC = createImage(320, 240);
imgD = createImage(320, 240);
img1 = createImage(320, 240);
img2 = createImage(320, 240);
img3 = createImage(320, 240);
img4 = createImage(320, 240);
}

function draw() {
background(255);

capture.loadPixels();
imgA.loadPixels();
imgB.loadPixels();
imgC.loadPixels();
imgD.loadPixels();
img1.loadPixels();
img2.loadPixels();
img3.loadPixels();
img4.loadPixels();


for (var y = 0; y < capture.height; y++) {
for (var x = 0; x < capture.width; x++) {
loc = (x + y * capture.width) * 4;

imgA.pixels[loc] = capture.pixels[loc];
imgA.pixels[loc + 1] = capture.pixels[loc + 1];
imgA.pixels[loc + 2] = capture.pixels[loc + 2];
imgA.pixels[loc + 3] = capture.pixels[loc + 3];

imgB.pixels[loc] = capture.pixels[loc];
imgB.pixels[loc + 1] = capture.pixels[loc + 1];
imgB.pixels[loc + 2] = capture.pixels[loc + 2];
imgB.pixels[loc + 3] = capture.pixels[loc + 3];

imgC.pixels[loc] = capture.pixels[loc];
imgC.pixels[loc + 1] = capture.pixels[loc + 1];
imgC.pixels[loc + 2] = capture.pixels[loc + 2];
imgC.pixels[loc + 3] = capture.pixels[loc + 3];

imgD.pixels[loc] = capture.pixels[loc];
imgD.pixels[loc + 1] = capture.pixels[loc + 1];
imgD.pixels[loc + 2] = capture.pixels[loc + 2];
imgD.pixels[loc + 3] = capture.pixels[loc + 3];

img1.pixels[loc] = capture.pixels[loc];
img1.pixels[loc + 1] = capture.pixels[loc + 1];
img1.pixels[loc + 2] = capture.pixels[loc + 2];
img1.pixels[loc + 3] = capture.pixels[loc + 3];

img2.pixels[loc] = capture.pixels[loc];
img2.pixels[loc + 1] = capture.pixels[loc + 1];
img2.pixels[loc + 2] = capture.pixels[loc + 2];
img2.pixels[loc + 3] = capture.pixels[loc + 3];

img3.pixels[loc] = capture.pixels[loc];
img3.pixels[loc + 1] = capture.pixels[loc + 1];
img3.pixels[loc + 2] = capture.pixels[loc + 2];
img3.pixels[loc + 3] = capture.pixels[loc + 3];

img4.pixels[loc] = capture.pixels[loc];
img4.pixels[loc + 1] = capture.pixels[loc + 1];
img4.pixels[loc + 2] = capture.pixels[loc + 2];
img4.pixels[loc + 3] = capture.pixels[loc + 3];
}

imgA.updatePixels();
imgB.updatePixels();
imgC.updatePixels();
imgD.updatePixels();
img1.updatePixels();
img2.updatePixels();
img3.updatePixels();
img4.updatePixels();

imgA.mask(imgMaskA);
imgB.mask(imgMask2);
imgC.mask(imgMaskC);
imgD.mask(imgMask4);

img1.mask(imgMaskA);
img2.mask(imgMask2);
img3.mask(imgMaskC);
img4.mask(imgMask4);

//IMAGE D, TOP ROW
push();
scale(-1,1);
image(imgD, -imgD.width, 0, 0, 0);
pop();

//IMAGE C, TOP ROW
image(imgC, 0, 0, 0, 0);

//IMAGE B, TOP ROW
push();
scale(-1,1);
image(imgB, -imgB.width, 0, 0, 0);
pop();

//IMAGE A, TOP ROW
image(imgA, 0, 0, 0, 0);

//IMAGE 1, TOP ROW
push();
scale(-1,1);
image(img1, -imgA.width, 0, -320, 0);
pop();

//IMAGE 2, TOP ROW
image(img2, 320, 0, 320, 0);

//IMAGE 3, TOP ROW
push();
scale(-1,1);
image(img3, -imgA.width, 0, -320, 0);
pop();

//IMAGE 4, TOP ROW
image(img4, 320, 0, 320, 0);

//FLIPPED FOR BOTTOM ROW
translate(640, 480);
rotate(PI);
}
}
